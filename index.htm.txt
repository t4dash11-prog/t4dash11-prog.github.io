<script>
function abrirSenha() {
  tela1.classList.add("hidden");
  tela2.classList.remove("hidden");
}

function verificar() {
  let senha = document.getElementById("senha").value;
  
  if (senha === "13/05/2024") {
    tela2.classList.add("hidden");
    tela3.classList.remove("hidden");
    musica.play();
    calcularTempo();
    escreverTexto();
  } else {
    erro.innerText = "Errou 😒 tenta de novo kkk";
  }
}

/* ⌨️ TEXTO DIGITANDO */
function escreverTexto() {
  const texto = "Desde que eu te conheci, tudo ficou mais leve...\n\nVocê é a pessoa que me faz rir até quando não quero 😂\n\nEu amo quando você briga comigo (mesmo estando errada 😌)\n\nAmo quando você rouba minha comida 😒❤️\n\nE principalmente… amo você sendo você.";

  let i = 0;
  const speed = 40;

  function digitar() {
    if (i < texto.length) {
      document.getElementById("texto").innerHTML += texto.charAt(i) === "\n" ? "<br>" : texto.charAt(i);
      i++;
      setTimeout(digitar, speed);
    }
  }

  document.getElementById("texto").innerHTML = "";
  digitar();
}

/* ⏳ CONTADOR */
function calcularTempo() {
  let inicio = new Date("2024-05-13");
  let hoje = new Date();
  let dias = Math.floor((hoje - inicio) / (1000 * 60 * 60 * 24));
  contador.innerText = dias + " dias ao seu lado ❤️";
}

function mostrarContador() {
  tela3.classList.add("hidden");
  telaContador.classList.remove("hidden");
}

function irGaleria() {
  telaContador.classList.add("hidden");
  tela4.classList.remove("hidden");
}

function final() {
  tela4.classList.add("hidden");
  tela5.classList.remove("hidden");
}
